# Stage 1: Build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy project files và restore
COPY *.csproj ./
RUN dotnet restore

# Copy toàn bộ source code
COPY . ./

# Build và publish ra /app
RUN dotnet publish -c Release -o /app

# Stage 2: Runtime
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app

# Copy build sang runtime
COPY --from=build /app .

# Mở port 80 để Render route HTTP
EXPOSE 80

# Chạy ứng dụng
ENTRYPOINT ["dotnet", "mskacenter.dll"]
